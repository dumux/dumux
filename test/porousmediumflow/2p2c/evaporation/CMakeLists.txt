# non-isothermal tests
dumux_add_test(NAME test_2p2cni_evaporation_box
              LABELS porousmediumflow 2p2c  2p2cni
              SOURCES main.cc

                               ${CMAKE_CURRENT_BINARY_DIR}/test_2p2cni_evaporation_constant_component_box-00010.vtu
